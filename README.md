### Maven dependency
```xml

    <repositories>
        <repository>
            <id>jitpack.io</id>
            <url>https://jitpack.io</url>
        </repository>
    </repositories>

    <dependencies>
        <dependency>
            <groupId>com.github.winlocker02</groupId>
            <artifactId>VkBotAPI</artifactId>
            <version>VERSION</version>
        </dependency>
    </dependencies>

```
